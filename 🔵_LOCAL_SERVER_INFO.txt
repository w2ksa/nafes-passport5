🌐 الروابط المحلية

⚡ الميزات الجديدة المتاحة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. سجلات التغييرات (Change History) ✅
   - زر: "سجلات التغييرات"
   - الموقع: لوحة التحكم
   - الوظيفة: عرض واستعادة التغييرات السابقة

2. النقاط الجماعية (Bulk Points) ✅
   - زر: "نقاط جماعية" (بنفسجي⚡)
   - الموقع: لوحة التحكم
   - الوظيفة: تحديث نقاط عدة طلاب دفعة واحدة


🔧 كيف تستخدم النقاط الجماعية
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. افتح: http://localhost:3000/admin
2. أدخل كود التحرير
3. اضغط زر "نقاط جماعية" (بنفسجي)
4. اختر:
   ✓ العملية: إضافة أو خصم
   ✓ المادة: عربي، رياضيات، علوم، إلخ
   ✓ عدد النقاط: مثلاً 10
   ✓ الطلاب: اضغط على الطالب لتحديده
5. اضغط "تطبيق على X طالب"


💡 نصائح
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

• لتحديد طالب: اضغط على صندوق الاختيار أو على اسم الطالب
• لتحديد الكل: اضغط زر "تحديد الكل" في الأعلى
• لإلغاء الكل: اضغط زر "إلغاء الكل"
• لتصفية حسب الصف: استخدم القائمة المنسدلة


🔄 التحديث التلقائي
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ Vite يحدّث الصفحة تلقائياً عند تعديل الكود
✅ Hot Module Replacement (HMR) مفعّل
✅ التغييرات تظهر فوراً


⚠️ إذا ما اشتغل
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. حدّث الصفحة بقوة:
   Mac: Cmd + Shift + R
   Windows/Linux: Ctrl + Shift + R

2. أعد تشغيل السيرفر:
   في Terminal: Ctrl + C
   ثم: pnpm dev

3. امسح الـ cache:
   في المتصفح: افتح Developer Tools → Application → Clear Storage
